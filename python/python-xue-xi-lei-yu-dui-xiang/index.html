面向对象介绍

类的成员

实例变量

实例变量就是对象个体特有的数据

#coding=utf-8

class Dog:
    def __init__(self,name,age):
        # 创建和初始化实例变量
        self.name=name 
        self.age=age

d = Dog('Dog',1)
print("我们家狗狗叫{0},{1}岁了。".format(d.name,d.age))

构造方法

类中的__init__()方法是一个非常特殊的方法，用来创建和初始化实例变量，这种方法就是构造方法。

#coding=utf-8

class Dog:
    def __init__(self,name,age,sex="雌性"):
        # 创建和初始化实例变量
        self.name=name 
        self.age=age
        self.sex=sex

d1 = Dog('Dog1',1)
d2 = Dog('Dog2',2,"雄性")
d3 = Dog(name='Dog2',sex="雄性",age=2)
print("{0}: {1}岁{2}}。".format(d1.name,d1.age,d1.sex))
print("{0}: {1}岁{2}}。".format(d2.name,d2.age,d2.sex))
print("{0}: {1}岁{2}}。".format(d3.name,d3.age,d3.sex))

实例方法

实例方法与实例变量一样，都是某个实例(对象)个人特有的方法

#coding=utf-8

class Dog:
    def __init__(self,name,age,sex="雌性"):
        # 创建和初始化实例变量
        self.name=name 
        self.age=age
        self.sex=sex
    # 实例方法
    def run(self):
        print("{}在跑".format(self.name))

    def speak(self,sound):
        print("{}在叫，'{}'!".format(self.name,sound))

d1 = Dog('Dog1',1)
d1.run()
d1.speak("汪汪汪")

类变量

类变量是属于类的变量，不属于单个对象;类变量是通过类名.类变量访问，实例变量是通过对象.实例变量访问

# coding=utf-8

class Account:
    interest_rate=0.0568 #利率

    def __init__(self,owner,amout):
        self.owner=owner
        self.amout=amout
    
account = Account("张三",1000000.0)
#实例变量是通过对象.实例变量访问
print("账户名：{0}".format(account.name))
print("账户金额：{0}".format(account.amout))
#类变量是通过类名.类变量访问
print("利率：{0}".format(Account.interest_rate))

类方法

类方法与类变量类似，属于类，不属于个体实例

# coding=utf-8

class Account:
    interest_rate=0.0668 #类变量利率

    def __init__(self,owner,amout):
        self.owner=owner
        self.amout=amout

    # 类方法
    @classmethod
    def interest_by(cls,amt):
        return cls.interest_rate*amt
    
interest = Account.interest_by(12000.0)
    
print("计算利息：{0:.4f}".format(interest))

封装性

私有变量

为了防止外部调用者随意存取类的内部数据(成员变量)，内部数据会被分装为私有变量；

私有变量：在变量前加上双下划线(_)

# coding=utf-8

class Account:
    __interest_rate=0.0568 #类私有变量利率

    def __init__(self,owner,amout):
        self.owner=owner #创建并初始化公有实例变量
        self.__amout=amout #创建并初始化私有实例变量__amout
    
    def desc(self):
        return "{0} 金额：{1} 利率：{2}".format(self.owner,self.__amout,Account.__interest_rate)
    
account = Account("张三",1000000.0)
account.desc()

print("账户名：{0}".format(account.name))
#print("账户金额：{0}".format(account.__amout)) # 会报错，私有变量不能访问
#print("利率：{0}".format(Account.__interest_rate)) # 会报错，私有类变量不能访问

私有方法

私有方法与私有变量的封装类似，在方法前加上双下划线(_)就是私有方法了

# coding=utf-8

class Account:
    __interest_rate=0.0568 #类私有变量利率

    def __init__(self,owner,amout):
        self.owner=owner #创建并初始化公有实例变量
        self.__amout=amout #创建并初始化私有实例变量__amout
    
    def __get_info(self):
        return "{0} 金额：{1} 利率：{2}".format(self.owner,self.__amout,Account.__interest_rate)

    def desc(self):
        print(self.__get_info())
    
account = Account("张三",1000000.0)
account.desc()

print("账户名：{0}".format(account.name))

使用属性

为了实现对象的封装，在一个类中不应该有公有的成员变量，这些成员变量应该被设计为私有的，然后通过公有的赋值(set)和取值(get)方法访问

#coding=utf-8

class Dog:
    def __init__(self,name,age,sex="雌性"):
        # 创建和初始化实例变量
        self.name=name 
        self.__age=age
        self.sex=sex
    # 实例方法
    def run(self):
        print("{}在跑".format(self.name))

    def speak(self,sound):
        print("{}在叫，'{}'!".format(self.name,sound))

    @property
    def age(self): # 替代get_age(self):
        return self.__age

    @age.setter
    def age(self,age): # 替代set_age(self,age)
        self.__age=age

dog = Dog("哈士奇",2)
print("{0}年龄：{1}".format(dog.name,dog.age))

dog.age=3 # dog.set_age(3)
print("修改{0}后年龄：{1}".format(dog.name,dog.age))

继承性

继承性也是面向对象重要的基本特性之一，在现实世界中继承关系无处不在

例如狗和动物的关系：狗是一种特殊的动物，具有动物的全部特性和行为，即数据和操作。在面向对象中动物是一般类，被称为父类；狗是特殊类，被称为子类。特殊类拥有一般类的全部数据和操作，可以称之为子类继承父类。

#coding=utf-8

class Animal:
    def __init__(self,name):
        self.name =name #实例变量name
    
    def show info(self):
        return"动物的名字:{0}".format(self.name)
    
    def move(self):
        print("动一动...")

class Cat(Animal):
    def __init__(self, name, age):
        super().__init__(name)
        self.age=age #实例变量age

cat =Cat('Tom'，2)
cat.move()
print(cat.show_info())

多继承

#coding=utf-8

class Horse:
    def __init__(self, name):
        self.name=name #实例变量name
    
    def show_info(self):
        return "马的名字:{0}".format(self.name)
    
    def run(self):
        print("马跑...")

class Donkey:
    def __init__(self, name):
        self.name =name #实例变量name
    
    def show_info(self):
        return "驴的名字:{0}".format(self.name)
    
    def run(self):
        print("驴跑...")
    
    def roll(self):
        print("驴打滚...")

class Mule(Horse, Donkey):
    def __init__(self, name, age):
        super()._init (name)
        self.age=age #实例变量age

# 继承的顺序是从左到右，左边的没有接着继续找下一个
m = Mule("骡子",2)
m.run() # 继承父类Hourse的方法
m.roll() # 继承父类Donkey的方法
print(m.show_info()) # 继承父类Hourse的方法

方法重写

#coding=utf-8

class Horse:
    def __init__(self, name):
        self.name=name #实例变量name
    
    def show_info(self):
        return "马的名字:{0}".format(self.name)
    
    def run(self):
        print("马跑...")

class Donkey:
    def __init__(self, name):
        self.name =name #实例变量name
    
    def show_info(self):
        return "驴的名字:{0}".format(self.name)
    
    def run(self):
        print("驴跑...")
    
    def roll(self):
        print("驴打滚...")

class Mule(Horse, Donkey):
    def __init__(self, name, age):
        super()._init (name)
        self.age=age #实例变量age

    def show_info(self):
        print("骡子：{0},年龄：{1}岁！".format(self.name,self.age))

# 继承的顺序是从左到右，左边的没有接着继续找下一个
m = Mule("骡子",2)
m.run() # 继承父类Hourse的方法
m.roll() # 继承父类Donkey的方法
print(m.show_info()) # 子类Mule的方法，重写了父类的show_info方法

多态性

多态指对象可以表现出多种形态

例如:猫、狗、鸭子都属于动物，它们都有叫和动等行为，但是叫的方式不同，动的方式也不动

继承与多态

在多个子类继承父类，并重写父类方法后，这些子类所创建的对象之间就是多态的，这些对象采用不同的方式实现父类方法

#coding=utf-8

class Animal:
    def speak(self):
        print("动物叫，但不知道是哪种动物！")

class Dog(Animal):
    def speak(self):
        print("小狗叫，汪汪汪！")

class Cat(Animal):
    def speak(self):
        print("小猫叫，喵喵喵！")
dog = Dog()
cat = Cat()
dog.speak()
cat.speak()

鸭子类型测试与多态

#coding-utf-8

# 设计一个函数start，接收具有speak方法的对象
def start(obj):
    obj.speak()

# 定义几个类都有speak方法
class Animal():
    def speak(self):
        print("动物叫，但不知道是哪种动物叫！")

class Dog(Animal):
    def speak(self):
        print("小狗叫，汪汪汪！")

class Cat(Animal):
    def speak(self):
        print("小猫叫，喵喵喵！")

class Car:
    def speak(self):
        print("汽车叫，滴滴滴！")

# start可以接收所有speak方法对象
start(Dog())
start(Cat())
start(Car())
