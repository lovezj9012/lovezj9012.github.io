数据库类配置

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace EFCoreDemo
{
    public class Book
    {
        public long Id { get; set; }

        public string Title { get; set; }

        public DateTime PublishDate { get; set; }

        public double Price { get; set; }

        public string AuthorName { get; set; }

        public int Age1 { get; set; }

        public int Age2 { get; set;}
    }
}

//Book的配置
using Microsoft.EntityFrameworkCore;
using Microsoft.EntityFrameworkCore.Metadata.Builders;

namespace EFCoreDemo
{

    /**
     * 该方式是FluentAPI,推荐使用
     * 缺点：复杂；优点：解耦
     */
    public class BookConfig : IEntityTypeConfiguration<Book>
    {
        public void Configure(EntityTypeBuilder<Book> builder)
        {
            //设置表名，也可以在实体类中使用属性设置表名
            builder.ToTable("T_Books");
            builder.Property(e=>e.Title).HasMaxLength(50).IsRequired();
            //Age2属性不映射到数据库
            builder.Ignore(e => e.Age2);
            //配置数据类型，HasColumnName配置数据列名，HasKey配置其他属性为主键
            builder.Property(e => e.AuthorName).HasColumnType("nvarchar(30)");
        }
    }
}

DbContext配置

using Microsoft.EntityFrameworkCore;
using Microsoft.Extensions.Logging;
using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Channels;
using System.Threading.Tasks;

namespace EFCoreDemo
{
    public class MyDbContext : DbContext
    {
        private readonly ILoggerFactory loggerFactory;

        public MyDbContext() { }
        public MyDbContext(ILoggerFactory loggerFactory)
        {
            this.loggerFactory = loggerFactory;
        }

        public DbSet<Book> books { get; set; }

        protected override void OnConfiguring(DbContextOptionsBuilder optionsBuilder)
        {
            base.OnConfiguring(optionsBuilder);
            MySqlServerVersion mysqlVersion = new MySqlServerVersion("5.7.0");
            string mysqlCon = "server=localhost;database=efcore;uid=root;pwd=root";
            optionsBuilder.UseMySql(mysqlCon, mysqlVersion);
            //1、标准日志 适用于排查故障
            //optionsBuilder.UseLoggerFactory(this.loggerFactory);

            //2、简单日志 适用于写测试代码
            optionsBuilder.LogTo(msg =>
            {
                //过滤掉无用的信息
                if (!msg.Contains("CommandExecuted"))
                {
                    return;
                }
                Console.WriteLine(msg);
            });

            //3、ToQueryString 适用于开发，只能用于返回类型为IQueryable的Where方法

        }

        protected override void OnModelCreating(ModelBuilder modelBuilder)
        {
            base.OnModelCreating(modelBuilder);
            modelBuilder.ApplyConfigurationsFromAssembly(this.GetType().Assembly);

            //也可以此方法中使用
            //modelBuilder.Entity<Book>().ToTable("T_Books");
        }
    }
}
